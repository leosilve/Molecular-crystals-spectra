75  // Integrate f(cx + d) with the given integration constants
76  static double IntegrateCore
77  (
78  const TFunctionObject& f,
79  double c, // slope of change of variables
80  double d, // intercept of change of variables
81  double targetAbsoluteError,
82  int& numFunctionEvaluations,
83  double& errorEstimate,
84  const double* abcissas,
85  const double* weights
86  )
87  {
88  targetAbsoluteError *= c;
89 
90  // Offsets to where each level's integration constants start.
91  // The last element is not a beginning but an end.
92  int offsets[] = {1, 4, 7, 13, 25, 49, 97, 193};
93  int numLevels = sizeof(offsets)/sizeof(int) - 1;
94 
95  double newContribution = 0.0;
96  double integral = 0.0;
97  errorEstimate = DBL_MAX;
98  double h = 1.0;
99  double previousDelta, currentDelta = DBL_MAX;
100 
101  integral = f(c*abcissas[0] + d) * weights[0];
102  int i;
103  for (i = offsets[0]; i != offsets[1]; ++i)
104  integral += weights[i]*(f(c*abcissas[i] + d) + f(-c*abcissas[i] + d));
105 
106  for (int level = 1; level != numLevels; ++level)
107  {
108  h *= 0.5;
109  newContribution = 0.0;
110  for (i = offsets[level]; i != offsets[level+1]; ++i)
111  newContribution += weights[i]*(f(c*abcissas[i] + d) + f(-c*abcissas[i] + d));
112  newContribution *= h;
113 
114  // difference in consecutive integral estimates
115  previousDelta = currentDelta;
116  currentDelta = fabs(0.5*integral - newContribution);
117  integral = 0.5*integral + newContribution;
118 
119  // Once convergence kicks in, error is approximately squared at each step.
120  // Determine whether we're in the convergent region by looking at the trend in the error.
121  if (level == 1)
122  continue; // previousDelta meaningless, so cannot check convergence.
123 
124  // Exact comparison with zero is harmless here. Could possibly be replaced with
125  // a small positive upper limit on the size of currentDelta, but determining
126  // that upper limit would be difficult. At worse, the loop is executed more
127  // times than necessary. But no infinite loop can result since there is
128  // an upper bound on the loop variable.
129  if (currentDelta == 0.0)
130  break;
131  double r = log( currentDelta )/log( previousDelta ); // previousDelta != 0 or would have been kicked out previously
132 
133  if (r > 1.9 && r < 2.1)
134  {
135  // If convergence theory applied perfectly, r would be 2 in the convergence region.
136  // r close to 2 is good enough. We expect the difference between this integral estimate
137  // and the next one to be roughly delta^2.
138  errorEstimate = currentDelta*currentDelta;
139  }
140  else
141  {
142  // Not in the convergence region. Assume only that error is decreasing.
143  errorEstimate = currentDelta;
144  }
145 
146  if (errorEstimate < 0.1*targetAbsoluteError)
147  break;
148  }
149 
150  numFunctionEvaluations = 2*i - 1;
151  errorEstimate *= c;
152  return c*integral;
153  }
